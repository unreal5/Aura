\chapter{Gameplay Effect}


\section{Gameplay Effect Stacking 总览}

\textbf{Stacking} 用来规定：当\emph{同一个 Gameplay Effect} 被重复应用时，GAS 如何处理
\textbf{层数、持续时间、周期、到期行为、溢出行为}。

\vspace{0.5em}
\textbf{可先记成一句话：}
\[
\boxed{
	\text{重复应用同一个 GE} \Rightarrow
	\text{是否并到同一份效果中？若并入，时间/周期/到期/溢出怎么处理？}
}
\]

% -------------------------------------------------
\subsubsection{1. EGameplayEffectStackingType}

\textbf{作用：}定义“重复应用的 GE 是否堆叠，以及按谁来堆叠”。

\begin{longtable}{@{}p{0.28\textwidth}p{0.24\textwidth}p{0.4\textwidth}@{}}
	\toprule
	\textbf{枚举值} & \textbf{显示名} & \textbf{含义} \\
	\midrule
	\endhead
	
	\texttt{None}
	& No Stacking
	& 不堆叠。每次应用都是\textbf{独立实例}。 \\
	
	\texttt{AggregateBySource}
	& Stack Per Source
	& 只有\textbf{同一个 Source} 反复施加时才会堆叠。 \\
	
	\texttt{AggregateByTarget}
	& Stack Per Target
	& 对同一个 Target 的重复应用会并入同一堆叠效果。 \\
	
	\bottomrule
\end{longtable}

\paragraph{记忆}
\begin{itemize}
	\item \texttt{None}：每次都是新效果
	\item \texttt{AggregateBySource}：\textbf{按施法者分堆}
	\item \texttt{AggregateByTarget}：\textbf{按目标分堆}
\end{itemize}

% -------------------------------------------------
\subsubsection{2. StackLimitCount}

\textbf{作用：}限制最大层数。

\[
\texttt{StackLimitCount} =
\begin{cases}
	\le 0, & \text{表示无限制} \\
	> 0, & \text{表示最大堆叠层数}
\end{cases}
\]

\paragraph{例子}
若
\[
\texttt{StackLimitCount} = 2
\]
则理论最大层数为 $2$。  
第 $3$ 次再应用时，就进入 \textbf{overflow（溢出）处理}。

\paragraph{记忆}
\[
\boxed{
	\texttt{StackLimitCount} = \text{最多能叠几层}
}
\]

% -------------------------------------------------
\subsubsection{3. EGameplayEffectStackingDurationPolicy}

\textbf{作用：}新 stack 成功加入后，\textbf{持续时间（Duration）}怎么处理。

\begin{longtable}{@{}p{0.30\textwidth}p{0.40\textwidth}p{0.30\textwidth}@{}}
	\toprule
	\textbf{枚举值} & \textbf{含义} & \textbf{记忆} \\
	\midrule
	\endhead
	
	\texttt{RefreshOnSuccessfulApplication}
	&
	每次成功叠层时，持续时间刷新为新的完整时长
	&
	重新计时
	\\
	
	\texttt{NeverRefresh}
	&
	层数变化，但当前剩余时间不变
	&
	只加层，不续时
	\\
	
	\texttt{ExtendDuration}
	&
	把新应用这一层的时长累加到当前剩余时间上
	&
	时间往后接
	\\
	
	\bottomrule
\end{longtable}

\paragraph{例子}
假设 GE 基础时长为 $5s$，当前剩余时间为 $2s$，现在又成功叠加一层：

\begin{itemize}
	\item \texttt{RefreshOnSuccessfulApplication}：
	\[
	2s \rightarrow 5s
	\]
	\item \texttt{NeverRefresh}：
	\[
	2s \rightarrow 2s
	\]
	\item \texttt{ExtendDuration}：
	\[
	2s \rightarrow 2s + 5s = 7s
	\]
\end{itemize}

% -------------------------------------------------
\subsubsection{4. EGameplayEffectStackingPeriodPolicy}

\textbf{作用：}如果这个 GE 是\textbf{周期效果（Periodic Effect）}，那么叠层时下一次 tick 的进度怎么处理。

\begin{longtable}{@{}p{0.3\textwidth}p{0.5\textwidth}p{0.2\textwidth}@{}}
	\toprule
	\textbf{枚举值} & \textbf{含义} & \textbf{记忆} \\
	\midrule
	\endhead
	
	\texttt{ResetOnSuccessfulApplication}
	&
	成功叠层后，通往下一次 tick 的进度被清空，重新计时
	&
	重置周期
	\\
	
	\texttt{NeverReset}
	&
	成功叠层后，周期进度不变，继续原来的 tick 节奏
	&
	不动 tick
	\\
	
	\bottomrule
\end{longtable}

\paragraph{例子}
若周期为 $1s$，当前已经走了 $0.8s$，此时叠层：

\begin{itemize}
	\item \texttt{ResetOnSuccessfulApplication}：重新从 $0s$ 开始计
	\item \texttt{NeverReset}：保持 $0.8s$ 进度，马上就会 tick
\end{itemize}

% -------------------------------------------------
\subsubsection{5. EGameplayEffectStackingExpirationPolicy}

\textbf{作用：}对于 \textbf{HasDuration} 类型 GE，当持续时间结束时，stack 怎么处理。

\begin{longtable}{@{}p{0.34\textwidth}p{0.42\textwidth}p{0.16\textwidth}@{}}
	\toprule
	\textbf{枚举值} & \textbf{含义} & \textbf{记忆} \\
	\midrule
	\endhead
	
	\texttt{ClearEntireStack}
	&
	效果到期时，整个 stack 一次性清掉
	&
	整堆清空
	\\
	
	\texttt{RemoveSingleStackAndRefreshDuration}
	&
	到期时只减去 $1$ 层，并刷新持续时间
	&
	掉一层并续时
	\\
	
	\texttt{RefreshDuration}
	&
	到期时只刷新持续时间，不自动减层
	&
	只续时
	\\
	
	\bottomrule
\end{longtable}

\paragraph{理解}
\begin{itemize}
	\item \texttt{ClearEntireStack}：最普通，时间到就没了
	\item \texttt{RemoveSingleStackAndRefreshDuration}：像“层层衰减”
	\item \texttt{RefreshDuration}：如果你自己在别处处理减层，它几乎能表现出“常驻刷新”
\end{itemize}

% -------------------------------------------------
\subsubsection{6. bFactorInStackCount}

\textbf{作用：}是否把层数计入 Modifier 的数值计算。

\begin{itemize}
	\item 若为 \texttt{true}：层数会影响最终 Modifier Magnitude
	\item 若为 \texttt{false}：层数只是一种状态，不自动放大数值
\end{itemize}

\paragraph{简单理解}
若单层伤害加成为 $10$，当前 stack 为 $3$：

\begin{itemize}
	\item \texttt{bFactorInStackCount = true}：可能按 $10 \times 3 = 30$
	\item \texttt{bFactorInStackCount = false}：仍按单层规则计算
\end{itemize}

\[
\boxed{
	\texttt{bFactorInStackCount} = \text{层数是否参与数值倍率}
}
\]

% -------------------------------------------------
\subsubsection{7. Overflow：超过上限时会发生什么}

当
$
\text{New Stack Count} > \texttt{StackLimitCount}
$
时，进入 \textbf{overflow} 处理。

相关字段有三组：

\begin{longtable}{@{}p{0.28\textwidth}p{0.58\textwidth}@{}}
	\toprule
	\textbf{字段} & \textbf{作用} \\
	\midrule
	\endhead
	
	\texttt{OverflowEffects}
	&
	发生 overflow 时额外施加的 Gameplay Effects。 \\
	\texttt{bDenyOverflowApplication}
	&
	若为 true，则在已达上限时，新的叠层尝试会被拒绝。 \\
	\texttt{bClearStackOnOverflow}
	&
	若为 true，且又设置了拒绝 overflow，则发生 overflow 时可把当前整堆清空。 \\
	\bottomrule
\end{longtable}

\paragraph{最关键结论}
设
$
\texttt{StackLimitCount} = 2
$
连续应用 3 次：

\begin{itemize}
	\item 第 1 次：成功，stack 变为 1
	\item 第 2 次：成功，stack 变为 2
	\item 第 3 次：进入 overflow 逻辑
\end{itemize}

此时：

\begin{enumerate}
	\item 若 \texttt{bDenyOverflowApplication = false}  
	则第 3 次\textbf{通常仍算成功}，但最终 stack 仍不会超过 2。
	
	\item 若 \texttt{bDenyOverflowApplication = true}  
	则第 3 次\textbf{会失败}。
	
	\item 若同时
	\[
	\texttt{bDenyOverflowApplication = true}
	\quad \text{且} \quad
	\texttt{bClearStackOnOverflow = true}
	\]
	则 overflow 时可能把现有 stack 整个清掉。
\end{enumerate}

\paragraph{一句话}
\[
\boxed{
	\text{超过上限后，不一定失败；是否失败取决于 } \texttt{bDenyOverflowApplication}
}
\]

\subsubsection{9. 最适合考试/面试/复习的总结}

\begin{itemize}
	\item \texttt{StackingType}：\textbf{按谁堆叠}
	\item \texttt{StackLimitCount}：\textbf{最多几层}
	\item \texttt{StackDurationRefreshPolicy}：\textbf{叠层时持续时间怎么变}
	\item \texttt{StackPeriodResetPolicy}：\textbf{叠层时周期 tick 怎么变}
	\item \texttt{StackExpirationPolicy}：\textbf{时间到后 stack 怎么变}
	\item \texttt{bFactorInStackCount}：\textbf{层数是否参与数值计算}
	\item \texttt{bDenyOverflowApplication}：\textbf{超上限时是否拒绝新应用}
	\item \texttt{bClearStackOnOverflow}：\textbf{超上限时是否清空整堆}
	\item \texttt{OverflowEffects}：\textbf{超上限时额外触发什么效果}
\end{itemize}

\[
\boxed{
	\text{GAS Stacking = 先决定是否并堆，再决定堆满后、续时、周期、到期怎么处理}
}
\]

\section{EGameplayEffectDurationType 与 Stacking 的关系}

\paragraph{核心问题}
在 GAS 中，先看一个 Gameplay Effect 是哪种 \texttt{DurationPolicy}，再判断它是否真正参与 \texttt{stacking}。  
因为：

\[
\boxed{
	\texttt{DurationPolicy} \text{ 决定 GE 是“瞬时执行”还是“作为 Active Effect 存在”}
}
\]

而 stacking 的本质是：  
\[
\boxed{
	\text{多个同类 GE 是否并入同一个 Active Gameplay Effect}
}
\]

所以二者关系非常紧密。

% -------------------------------------------------
\subsubsection{1. EGameplayEffectDurationType}

\begin{longtable}{@{}p{0.26\textwidth}p{0.2\textwidth}p{0.5\textwidth}@{}}
	\toprule
	\textbf{枚举值} & \textbf{含义} & \textbf{直观理解} \\
	\midrule
	\endhead
	
	\texttt{Instant}
	&
	立即执行
	&
	只在应用瞬间生效，不作为持续中的 Active Effect 保留
	\\
	
	\texttt{Infinite}
	&
	无限持续
	&
	一旦应用，会一直留在 Active Effects 中，直到被移除
	\\
	
	\texttt{HasDuration}
	&
	有时长
	&
	应用后保留一段时间，直到持续时间结束
	\\
	
	\bottomrule
\end{longtable}

\paragraph{一句话}
\[
\texttt{Instant} = 执行后就结束
\qquad
\texttt{Infinite} = 一直挂着
\qquad
\texttt{HasDuration} = 挂一段时间
\]

% -------------------------------------------------
\subsubsection{2. 为什么 DurationPolicy 会影响 Stacking}

GAS 源码里对 \texttt{EGameplayEffectStackingType} 的注释是：

\[
\text{stacking 作用于 ``non-instant GE's''}
\]

也就是说，\textbf{stacking 主要针对非 Instant 的 GE}，因为只有它们会真正进入
\texttt{ActiveGameplayEffectsContainer}。

\paragraph{关键理解}
\begin{itemize}
	\item \texttt{Instant}：通常\textbf{不形成可持续存在的 stack}
	\item \texttt{Infinite}：可以 stack
	\item \texttt{HasDuration}：可以 stack
\end{itemize}

所以：

\[
\boxed{
	\texttt{Instant} \text{ 一般不看“持续中的 stack”} ,
	\quad
	\texttt{Infinite / HasDuration} \text{ 才是真正的 stacking 主战场}
}
\]

	\item 跟“\textbf{层数、overflow、数值倍率}”有关的字段：主要看 \texttt{Infinite / HasDuration}
\end{itemize}

% -------------------------------------------------
\subsubsection{5. 分类型理解}

\paragraph{\texttt{Instant}}
\begin{itemize}
	\item 它的本质是：\textbf{执行一次就结束}
	\item 一般不会作为持续中的 Active Effect 留在容器里
	\item 因此通常不讨论“剩余时间刷新”“到期掉一层”等 stacking 行为
	\item 重复应用更像：
	\[
	\text{执行一次} + \text{再执行一次} + \text{再执行一次}
	\]
\end{itemize}

\paragraph{\texttt{Infinite}}
\begin{itemize}
	\item 会进入 Active Effects
	\item 可以形成 stack
	\item 但因为它\textbf{没有自然倒计时结束}，所以：
	\begin{itemize}
		\item \texttt{StackDurationRefreshPolicy} 基本没有实际意义
		\item \texttt{StackExpirationPolicy} 基本也不重要
	\end{itemize}
	\item 更重要的是：
	\begin{itemize}
		\item \texttt{StackLimitCount}
		\item overflow 行为
		\item \texttt{bFactorInStackCount}
		\item 若有 periodic，则 \texttt{StackPeriodResetPolicy}
	\end{itemize}
\end{itemize}

\paragraph{\texttt{HasDuration}}
\begin{itemize}
	\item 是 stacking 规则最完整、最典型的使用场景
	\item 会进入 Active Effects
	\item 有明确剩余时间
	\item 因而下面这些都很重要：
	\begin{itemize}
		\item \texttt{StackDurationRefreshPolicy}
		\item \texttt{StackExpirationPolicy}
		\item \texttt{StackLimitCount}
		\item overflow 行为
		\item 若有 periodic，则 \texttt{StackPeriodResetPolicy}
	\end{itemize}
\end{itemize}

% -------------------------------------------------
\subsubsection{6. 例子：三种 DurationPolicy 的重复应用区别}

假设连续应用同一个 GE 三次。

\paragraph{\texttt{Instant}}
更像：
\[
\text{执行}_1 + \text{执行}_2 + \text{执行}_3
\]
而不是：
\[
\text{一个持续存在的效果} \; \text{stack 到 3 层}
\]

\paragraph{\texttt{Infinite}}
更像：
\[
1 \text{层} \rightarrow 2 \text{层} \rightarrow 3 \text{层}
\]
直到遇到 \texttt{StackLimitCount}。

\paragraph{\texttt{HasDuration}}
不仅可能：
\[
1 \text{层} \rightarrow 2 \text{层} \rightarrow 3 \text{层}
\]
还要继续考虑：
\begin{itemize}
	\item 新层加入后剩余时间是否刷新
	\item periodic tick 是否重置
	\item 到期时是清空整堆还是掉一层
\end{itemize}

% -------------------------------------------------
\subsubsection{7. 最容易混的特殊点：Instant Prediction}

\textbf{例外：}客户端预测时，源码里会把 \texttt{Instant} 临时当成类似 ``Infinite Duration'' 来处理，
目的是让预测结果先挂住，之后再由服务端校正。

\paragraph{笔记版理解}
\[
\boxed{
	\texttt{Instant} \text{ 在正常逻辑下不是真正的持续 stack；
		但在预测场景下，可能被临时当作可持续效果处理。}
}
\]

\paragraph{因此}
\begin{itemize}
	\item 理论总结时：把 \texttt{Instant} 记成“不真正参与持续 stacking”
	\item 看联机预测细节时：要知道它有“临时持续化”的特殊处理
\end{itemize}

\subsubsection{9. 背诵版总结}

\begin{itemize}
	\item \texttt{Instant}：\textbf{重复应用 = 重复执行}，通常不看持续 stacking
	\item \texttt{Infinite}：\textbf{可以 stack}，但通常不关心 duration refresh / expiration
	\item \texttt{HasDuration}：\textbf{最完整的 stacking 场景}，要同时看层数、剩余时间、周期、到期、overflow
\end{itemize}

\[
\boxed{
	\texttt{Instant} \Rightarrow \text{执行型}
	\qquad
	\texttt{Infinite} \Rightarrow \text{常驻型 stacking}
	\qquad
	\texttt{HasDuration} \Rightarrow \text{限时型 stacking}
}
\]



\subsection{EGameplayEffectStackingExpirationPolicy}

\textbf{作用：}  
用于规定：当一个\textbf{可堆叠且有持续时间的 Gameplay Effect 到期}时，stack 应该如何处理。

\vspace{0.5em}
\textbf{它回答的问题是：}
\[
\text{“当这个 GE 的持续时间走完后，这一堆 stack 怎么办？”}
\]

\begin{longtable}{@{}p{0.34\textwidth}p{0.40\textwidth}p{0.18\textwidth}@{}}
	\toprule
	\textbf{枚举值} & \textbf{含义} & \textbf{记忆} \\
	\midrule
	\endhead
	
	\texttt{ClearEntireStack}
	&
	效果到期时，\textbf{整个 stack 一次性清空}
	&
	整堆消失
	\\
	
	\texttt{RemoveSingleStackAndRefreshDuration}
	&
	效果到期时，\textbf{只减少 1 层}，然后\textbf{刷新持续时间}
	&
	掉一层并重新计时
	\\
	
	\texttt{RefreshDuration}
	&
	效果到期时，\textbf{不自动减层}，只\textbf{刷新持续时间}
	&
	只续时，不掉层
	\\
	
	\bottomrule
\end{longtable}

\paragraph{直观例子}
假设某个 GE：
\begin{itemize}
	\item 当前 stack 数为 $3$
	\item 持续时间为 $5s$
	\item 现在时间到了
\end{itemize}

那么：

\begin{itemize}
	\item \texttt{ClearEntireStack}
	\[
	3 \rightarrow 0
	\]
	
	\item \texttt{RemoveSingleStackAndRefreshDuration}
	\[
	3 \rightarrow 2
	\]
	然后持续时间重新开始计时。
	
	\item \texttt{RefreshDuration}
	\[
	3 \rightarrow 3
	\]
	层数不变，只是持续时间重新开始计时。
\end{itemize}

\paragraph{使用理解}
\begin{itemize}
	\item \texttt{ClearEntireStack}：普通限时 Buff，时间到就全部结束
	\item \texttt{RemoveSingleStackAndRefreshDuration}：适合“逐层衰减”效果
	\item \texttt{RefreshDuration}：适合你\textbf{自己用别的逻辑控制减层}的情况
\end{itemize}

\paragraph{注意}
这个策略主要对
\[
\texttt{DurationPolicy = HasDuration}
\]
最有意义。因为只有\textbf{会到期}的 GE，才需要讨论“到期后 stack 怎么处理”。

		\node[box, below left=of expire] (clear) {\texttt{ClearEntireStack}\\3 $\rightarrow$ 0};

		\node[box, below right=of expire] (refresh) {\texttt{RefreshDuration}\\3 $\rightarrow$ 3，只重置时间};

\paragraph{一句话总结}
\[
\boxed{
	\texttt{ClearEntireStack} = 全清
	\quad
	\texttt{RemoveSingleStackAndRefreshDuration} = 掉一层并续时
	\quad
	\texttt{RefreshDuration} = 只续时
}
\]

\subsection{EGameplayEffectStackingPeriodPolicy}

\textbf{作用：}  
用于规定：当一个\textbf{带 Period 的可堆叠 Gameplay Effect} 成功叠层时，  
\textbf{距离下一次周期触发（tick）的进度}要如何处理。

\vspace{0.5em}
\textbf{它回答的问题是：}
\[
\text{“叠层之后，下一个周期 tick 要重新计时，还是沿用原来的进度？”}
\]

\begin{longtable}{@{}p{0.25\textwidth}p{0.50\textwidth}p{0.18\textwidth}@{}}
	\toprule
	\textbf{枚举值} & \textbf{含义} & \textbf{记忆} \\
	\midrule
	\endhead
	
	\texttt{ResetOnSuccessfulApplication}
	&
	每次\textbf{成功叠层}后，通往下一次 tick 的进度\textbf{清空并重新开始计时}
	&
	重置周期
	\\
	
	\texttt{NeverReset}
	&
	即使成功叠层，当前距离下一次 tick 的进度也\textbf{保持不变}
	&
	保持原节奏
	\\
	
	\bottomrule
\end{longtable}

\paragraph{直观例子}
假设某个 GE：
\begin{itemize}
	\item Period = $1s$
	\item 当前已经过去了 $0.8s$
	\item 此时又成功叠加了一层
\end{itemize}

那么：

\begin{itemize}
	\item \texttt{ResetOnSuccessfulApplication}
	\[
	0.8s \rightarrow 0s
	\]
	即：重新开始等下一个 tick。
	
	\item \texttt{NeverReset}
	\[
	0.8s \rightarrow 0.8s
	\]
	即：保留原本进度，很快就会 tick。
\end{itemize}

\paragraph{使用理解}
\begin{itemize}
	\item \texttt{ResetOnSuccessfulApplication}：适合“每次补上一层，就重新计周期”
	\item \texttt{NeverReset}：适合“叠层不打乱原本 tick 节奏”
\end{itemize}

\paragraph{注意}
这个策略只有在 GE \textbf{本身带有 Period} 时才有意义。  
如果没有周期触发，就没有“下一次 tick 进度”可言。

\paragraph{一图记忆}
\begin{center}
	\begin{tikzpicture}[
		node distance=8mm and 10mm,
		box/.style={draw, rounded corners, align=center, minimum width=36mm, minimum height=8mm},
		arrow/.style={-Latex, thick}
		]
		\node[box] (stack) {叠层时，当前 Tick 进度 = 0.8s};
		
		\node[box, below left=of stack] (reset) {\texttt{ResetOnSuccessfulApplication}\\0.8 $\rightarrow$ 0};
		\node[box, below right=of stack] (keep) {\texttt{NeverReset}\\0.8 $\rightarrow$ 0.8};
		
		\draw[arrow] (stack) -- (reset);
		\draw[arrow] (stack) -- (keep);
	\end{tikzpicture}
\end{center}

\paragraph{一句话总结}
\[
\boxed{
	\texttt{ResetOnSuccessfulApplication} = 叠层后重置周期
	\quad
	\texttt{NeverReset} = 叠层后保持原 tick 进度
}
\]

\subsection{EGameplayEffectPeriodInhibitionRemovedPolicy}

\textbf{作用：}  
用于规定：当一个\textbf{带 Period 的 Gameplay Effect} 先前因为某种原因被\textbf{抑制（inhibited）}，  
而现在\textbf{抑制被解除}时，接下来的 \textbf{period tick} 应该怎么处理。

\vspace{0.5em}
\textbf{它回答的问题是：}
\[
\text{“周期效果恢复工作后，下一次 tick 什么时候发生？”}
\]

\paragraph{什么是 Inhibition}
可以简单理解为：  
这个 GE 还在，但它的周期执行暂时被\textbf{暂停 / 压住}了。  
当 inhibition removed 时，GE 又恢复正常工作。

\begin{longtable}{@{}p{0.17\textwidth}p{0.7\textwidth}p{0.13\textwidth}@{}}
	\toprule
	\textbf{枚举值} & \textbf{含义} & \textbf{记忆} \\
	\midrule
	\endhead
	
	\texttt{NeverReset}
	&
	不重置周期计时。恢复后，继续按原本的 period 节奏走，好像 inhibition 没发生过一样。
	&
	继续原节奏
	\\
	
	\texttt{ResetPeriod}
	&
	重置周期。恢复后，从此刻开始重新计一个完整的 period，下一次 tick 在\textbf{一个完整周期之后}发生。
	&
	重新计时
	\\
	
	\texttt{ExecuteAndResetPeriod}
	&
	恢复时\textbf{立刻执行一次}，然后再从此刻开始重置 period。
	&
	先立刻触发，再重新计时
	\\
	
	\bottomrule
\end{longtable}

\paragraph{直观例子}
假设：
\begin{itemize}
	\item Period = $1s$
	\item GE 在距离下一次 tick 还差 $0.2s$ 时被 inhibition
	\item 过一会 inhibition 被移除
\end{itemize}

那么：

\begin{itemize}
	\item \texttt{NeverReset}：  
	恢复后继续原进度，所以\textbf{再过 $0.2s$ 就 tick}。
	
	\item \texttt{ResetPeriod}：  
	恢复后\textbf{重新开始计 $1s$}，所以下一次 tick 在 $1s$ 后。
	
	\item \texttt{ExecuteAndResetPeriod}：  
	恢复时\textbf{立刻 tick 一次}，然后下一次再等完整的 $1s$。
\end{itemize}

\paragraph{使用理解}
\begin{itemize}
	\item \texttt{NeverReset}：保留原来的 tick 节奏
	\item \texttt{ResetPeriod}：恢复后重新开始一个完整周期
	\item \texttt{ExecuteAndResetPeriod}：恢复时补一次效果，然后进入新周期
\end{itemize}

\paragraph{一图记忆}
\begin{center}
	\begin{tikzpicture}[
		node distance=8mm and 10mm,
		box/.style={draw, rounded corners, align=center, minimum width=36mm, minimum height=8mm},
		arrow/.style={-Latex, thick}
		]
		\node[box] (start) {Inhibition 被移除};
		
		\node[box, below left=of start] (never) {\texttt{NeverReset}\\继续原 tick 进度};
		\node[box, below=of start] (reset) {\texttt{ResetPeriod}\\重新开始一个完整周期};

\[
\boxed{
	\texttt{NeverReset} = 接着走
	\quad
	\texttt{ResetPeriod} = 重新计
	\quad
	\texttt{ExecuteAndResetPeriod} = 先触发一次再重新计
}
\]

\section{Gameplay Effect：Duration / Period / Stacking 总复习}

\paragraph{总思路}
学习一个 Gameplay Effect 时，可以按下面顺序理解：

\[
\boxed{
	\text{先看 DurationPolicy}
	\;\rightarrow\;
	\text{再看是否有 Period}
	\;\rightarrow\;
	\text{最后看是否启用 Stacking}
}
\]

% -------------------------------------------------
\subsubsection{1. 第一步：看 DurationPolicy}

\begin{longtable}{@{}p{0.22\textwidth}p{0.30\textwidth}p{0.34\textwidth}@{}}
	\toprule
	\textbf{DurationPolicy} & \textbf{含义} & \textbf{重点} \\
	\midrule
	\endhead
	
	\texttt{Instant}
	&
	立即执行后结束
	&
	通常不形成持续中的 Active Effect
	\\
	
	\texttt{Infinite}
	&
	持续存在，直到被手动移除
	&
	适合常驻 Buff / Debuff
	\\
	
	\texttt{HasDuration}
	&
	持续一段时间后结束
	&
	适合限时 Buff / Debuff
	\\
	
	\bottomrule
\end{longtable}

\paragraph{一句话}
\[
\texttt{Instant} = 执行型
\qquad
\texttt{Infinite} = 常驻型
\qquad
\texttt{HasDuration} = 限时型
\]

% -------------------------------------------------
\subsubsection{2. 第二步：看有没有 Period}

\textbf{Period} 表示这个 GE 是否会按固定间隔重复触发。

\begin{itemize}
	\item 没有 Period：不是周期效果
	\item 有 Period：会每隔一段时间 tick 一次
\end{itemize}

\paragraph{相关字段}
\begin{longtable}{@{}p{0.34\textwidth}p{0.50\textwidth}@{}}
	\toprule
	\textbf{字段 / 策略} & \textbf{作用} \\
	\midrule
	\endhead
	
	\texttt{Period}
	&
	定义多久 tick 一次
	\\
	
	\texttt{bExecutePeriodicEffectOnApplication}
	&
	应用成功时是否先立刻执行一次
	\\
	
	\texttt{EGameplayEffectStackingPeriodPolicy}
	&
	叠层时，当前周期进度是否重置
	\\
	
	\texttt{EGameplayEffectPeriodInhibitionRemovedPolicy}
	&
	inhibition 解除后，下一次 tick 如何安排
	\\
	
	\bottomrule
\end{longtable}

\paragraph{一句话}
\[
\boxed{
	\texttt{Period} = 间隔
	\quad
	\texttt{ExecuteOnApplication} = 首跳
	\quad
	\texttt{StackingPeriodPolicy} = 叠层后的节奏
	\quad
	\texttt{InhibitionRemovedPolicy} = 恢复后的节奏
}
\]

% -------------------------------------------------
\subsubsection{3. 第三步：看是否启用 Stacking}

\textbf{Stacking} 关注的是：重复应用同一个 GE 时，是否并入同一个 stack，以及并入后如何处理。

\begin{longtable}{@{}p{0.34\textwidth}p{0.50\textwidth}@{}}
	\toprule
	\textbf{字段 / 策略} & \textbf{作用} \\
	\midrule
	\endhead
	
	\texttt{StackingType}
	&
	按谁堆叠：不堆叠 / 按 Source / 按 Target
	\\
	
	\texttt{StackLimitCount}
	&
	最大层数
	\\
	
	\texttt{StackDurationRefreshPolicy}
	&
	叠层时剩余时间怎么变化
	\\
	
	\texttt{StackPeriodResetPolicy}
	&
	叠层时周期进度是否重置
	\\
	
	\texttt{StackExpirationPolicy}
	&
	到期时整堆怎么处理
	\\
	
	\texttt{bFactorInStackCount}
	&
	层数是否参与数值计算
	\\
	
	\texttt{bDenyOverflowApplication}
	&
	超出上限时是否拒绝新的应用
	\\
	
	\texttt{bClearStackOnOverflow}
	&
	overflow 时是否清空整堆
	\\
	
	\texttt{OverflowEffects}
	&
	overflow 时额外施加什么效果
	\\
	
	\bottomrule
\end{longtable}

\subsubsection{4. DurationPolicy 与 Stacking 的关系}

\begin{longtable}{@{}p{0.20\textwidth}p{0.18\textwidth}p{0.18\textwidth}p{0.30\textwidth}@{}}
	\toprule
	\textbf{DurationPolicy} & \textbf{能否持续存在} & \textbf{是否真正适合 Stacking} & \textbf{说明} \\
	\midrule
	\endhead

	&
	通常否

	\\
	
	\texttt{Infinite}
	&
	是
	&
	是
	&
	可以叠层，但一般不讨论“到期”
	\\
	
	\texttt{HasDuration}
	&
	是
	&
	是

	\\
	
	\bottomrule
\end{longtable}

\paragraph{最关键结论}
\[
\boxed{
	\texttt{Instant} \text{ 更像“重复执行”}
	\quad
	\texttt{Infinite / HasDuration} \text{ 才是真正的 stacking 主场}
}
\]

% -------------------------------------------------
\subsubsection{5. 三个最常用 stacking policy}

\paragraph{(1) Duration Policy}
叠层成功后，\textbf{剩余时间}怎么处理：

\begin{itemize}
	\item \texttt{RefreshOnSuccessfulApplication}：重置时间
	\item \texttt{NeverRefresh}：时间不变
	\item \texttt{ExtendDuration}：剩余时间累加
\end{itemize}

\paragraph{(2) Period Policy}
叠层成功后，\textbf{周期进度}怎么处理：

\begin{itemize}
	\item \texttt{ResetOnSuccessfulApplication}：重置周期
	\item \texttt{NeverReset}：保持原节奏
\end{itemize}

\paragraph{(3) Expiration Policy}
当 GE \textbf{到期}时，stack 怎么处理：

\begin{itemize}
	\item \texttt{ClearEntireStack}：整堆清空
	\item \texttt{RemoveSingleStackAndRefreshDuration}：掉一层并续时
	\item \texttt{RefreshDuration}：只续时，不掉层
\end{itemize}

\begin{itemize}
	\item \texttt{DurationPolicy}：先判断它是执行型、常驻型还是限时型
